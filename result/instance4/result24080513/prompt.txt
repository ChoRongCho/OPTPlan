    # Preconditions
    if self.robot_handempty and not obj.in_box and not obj.out_box:

def place(self, obj, bin):
    # Preconditions
    if self.robot_now_holding == obj and obj.out_box:
        # Effects
        self.state_handempty()
        obj.in_box = True
        obj.out_box = False
        bin.in_bin_objects.append(obj.index)
        print(f"Place {obj.name} in {bin.name}")
    else:
        print(f"Cannot Place {obj.name}")

        # Effects
        obj.pushed = True
        print(f"Push {obj.name}")
    else:

def fold(self, obj, bin):
    # Preconditions
