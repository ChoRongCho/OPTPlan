        self.goal = goal

            obj.is_packed = True
            obj.is_outside = False
            self.state_handempty()
        else:
            print(f"Cannot Place {obj.name}")

            print(f"Push {obj.name}")

            print(f"Cannot Push {obj.name}")

        if self.robot_handempty and obj.is_foldable and obj.is_outside:

            obj.is_outside = True
